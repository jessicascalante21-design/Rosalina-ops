function checkShiftClose() {
  const h = new Date().getHours(), m = new Date().getMinutes();
  const isAMClose = (h === 5 && m < 10);
  const isPMClose = (h === 2 && m < 10);
  if (isAMClose || isPMClose) buildShiftSummary(isAMClose ? 'AM Shift (8:00 AM – 5:00 PM)' : 'PM Shift (5:00 PM – 2:00 AM)');
}

function buildShiftSummary(shiftName) {
  const now = new Date();
  document.getElementById('shiftTitle').textContent = `${shiftName} — Closing summary`;
  document.getElementById('shiftDate').textContent  = now.toLocaleDateString('en-US', {weekday:'long',month:'long',day:'numeric'});
  document.getElementById('ss_shift').textContent   = shiftName;
  document.getElementById('ss_interactions').textContent = session.requests.length + session.calls.length;
  document.getElementById('ss_requests').textContent    = session.requests.length;
  document.getElementById('ss_calls').textContent       = session.calls.length;
  document.getElementById('ss_prearr').textContent      = session.prearr.length;

  const urgent = session.requests.filter(r => r.urgency === 'high');
  document.getElementById('ss_open').textContent = urgent.length > 0
    ? urgent.map(r => `${r.room} — ${r.type}`).join(', ')
    : 'None 🎉';

  // AI summary generation
  const ai = generateAISummary(shiftName, session);
  document.getElementById('aiSummary').textContent = ai;

  showView('shift');
}

function generateAISummary(shift, s) {
  const reqs = s.requests.length;
  const calls = s.calls.length;
  const urgent = s.requests.filter(r => r.urgency === 'high');
  const types = [...new Set(s.requests.map(r => r.type))];

  let text = `${shift} completed. `;
  if (reqs === 0 && calls === 0) {
    text += 'Quiet shift — no guest interactions recorded. All systems normal.';
  } else {
    text += `${reqs + calls} total guest interaction${(reqs+calls)!==1?'s':''} — ${calls} video call${calls!==1?'s':''} and ${reqs} service request${reqs!==1?'s':''}.`;
    if (types.length) text += ` Most common requests: ${types.slice(0,3).join(', ')}.`;
    if (urgent.length) text += ` ⚠️ ${urgent.length} urgent item${urgent.length!==1?'s':''} pending for next shift: ${urgent.map(r=>r.room).join(', ')}.`;
    if (s.prearr.length) text += ` ${s.prearr.length} pre-arrival form${s.prearr.length!==1?'s':''} received — coordinate with housekeeping.`;
    text += ' All interactions logged.';
  }
  return text;
}
